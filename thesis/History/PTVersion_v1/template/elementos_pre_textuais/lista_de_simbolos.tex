\chapter*{Lista de Símbolos}
\pdfbookmark[0]{Lista de Símbolos}{simbolos}

\begin{description}

\item[$x_{d,t}$] Motorista \(d\) trabalha no período \(t\)
\item[$r_{d,t}$] Motorista \(d\) está em descanso no período \(t\)
\item[$h_{d,t}$] Condução acumulada no período \(t\)
\item[$s_{d,t}$] Início de jornada no período \(t\)
\item[$z_{d,w}$] Uso de extensão diária na semana \(w\)
\item[$\Delta$] Duração de cada período
\item[$L^{\text{dia}}$] Limite diário de condução
\item[$L^{\text{ext}}$] Extensão diária permitida
\item[$L^{\text{sem}}$] Limite semanal de condução
\item[$L^{14d}$] Limite de condução em 14 dias
\item[$R_{\text{dia}}$] Descanso diário normal
\item[$R_{\text{dia-red}}$] Descanso diário reduzido
\item[$R_{\text{sem}}$] Descanso semanal
\item[$P_{\text{pausa}}$] Limite de condução contínua antes da pausa

\end{description}
